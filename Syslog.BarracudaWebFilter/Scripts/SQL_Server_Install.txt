-- =============================================
-- Author:		Andrew Draut
-- Create date: 3/12/2010
-- Description:	Inserts a log message from a Barracuda Web Filter
-- =============================================
CREATE PROCEDURE [dbo].[InsertBarracudaWebFilterMessage]
	-- Add the parameters for the stored procedure here
	@MsgDateTime datetime,
	@SourceIP varchar(15),
	@DestIPOrDns varchar(64),
	@ContentType varchar(64),
	@URL varchar(256),
	@Action varchar(10),
	@Reason varchar(10),
	@FormatVersion char(1),
	@MatchFlag char(1),
	@TQFlag char(1),
	@ActionType char(1),
	@SrcType char(1),
	@SrcDetail varchar(64),
	@DestType char(1),
	@SpyType char(1),
	@MatchedPart varchar(64),
	@MatchedCategory varchar(128),
	@LocalUser varchar(32)
AS
BEGIN
    -- Insert statements for procedure here
	INSERT INTO dbo.BarracudaWebFilter
	(MsgDateTime,
	SourceIP,
	DestIPOrDns,
	ContentType,
	URL,
	Action,
	Reason,
	FormatVersion,
	MatchFlag,
	TQFlag,
	ActionType,
	SrcType,
	SrcDetail,
	DestType,
	SpyType,
	MatchedPart,
	MatchedCategory,
	LocalUser)
	VALUES
	(@MsgDateTime,
	@SourceIP,
	@DestIPOrDns,
	@ContentType,
	@URL,
	@Action,
	@Reason,
	@FormatVersion,
	@MatchFlag,
	@TQFlag,
	@ActionType,
	@SrcType,
	@SrcDetail,
	@DestType,
	@SpyType,
	@MatchedPart,
	@MatchedCategory,
	@LocalUser)
END

GO

-- =============================================
-- Author:		Andrew Draut
-- Create date: 5/27/2010
-- Description:	Inserts a log message from a Barracuda Spam Filter
-- =============================================
CREATE PROCEDURE [dbo].[InsertBarracudaSpamFilterMessage]
	-- Add the parameters for the stored procedure here
	@MsgDateTime datetime
   ,@IP varchar(75)
   ,@ID varchar(50)
   ,@StartTime bigint
   ,@EndTime bigint
   ,@Encryption varchar(50)
   ,@Sender varchar(100)
   ,@Recipient varchar(100)
   ,@Score float
   ,@Action int
   ,@Reason int
   ,@ReasonExtra varchar(100)
   ,@Subject varchar(250)
AS
BEGIN
    -- Insert statements for procedure here
	INSERT INTO [Syslog].[dbo].[BarracudaSpamFilter]
	   ([MsgDateTime]
	   ,[IP]
	   ,[ID]
	   ,[StartTime]
	   ,[EndTime]
	   ,[Encryption]
	   ,[Sender]
	   ,[Recipient]
	   ,[Score]
	   ,[Action]
	   ,[Reason]
	   ,[ReasonExtra]
	   ,[Subject])
     VALUES
       (@MsgDateTime
       ,@IP
       ,@ID
       ,@StartTime
       ,@EndTime
       ,@Encryption
       ,@Sender
       ,@Recipient
       ,@Score
       ,@Action
       ,@Reason
       ,@ReasonExtra
       ,@Subject)
END

GO
